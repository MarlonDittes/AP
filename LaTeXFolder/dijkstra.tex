\begin{algorithm}
\captionsetup{labelformat=empty, labelsep=none}
\caption{Dijkstra's Algorithm (G = (V, E), s, t)}
\begin{algorithmic}[2]
    \State $dist[s] = 0$
    \State $Q.clear(), Q.add(s,0)$
    \While{$!Q.empty()$}
        \State $u \gets Q.deleteMin()$
        \For{all $edges$ e = (u,v) $\in E$}
            \If{$dist[u] + len(e) < dist[v]$}
                \State $dist[v] \gets dist[u] + len(e)$
                \If{$v \in Q$} \State $Q.decreaseKey(v, d[v])$
                \Else{} \State $Q.insert(v, d[v])$
                \EndIf
            \EndIf
        \EndFor
    \EndWhile
\end{algorithmic}
\end{algorithm}
